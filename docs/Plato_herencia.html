def huella_nutricional

	@huella_energetica

	if @valor <= 670
		@huella_energetica = 1.0
	elsif @valor > 670 && @valor <= 830
		@huella_energetica = 2.0
	else  
		@huella_energetica = 3.0
	end

	@huella_carbono

	if @total_ve <= 800
		@huella_carbono = 1.0
	elsif @total_ve > 800 && @total_ve <= 1200
		@huella_carbono = 2.0
	else 
		@huella_carbono = 3.0
	end

	return (@huella_energetica+@huella_carbono)/2

end
