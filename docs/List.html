def extract_head
    aux=""
    if (@head == nil)
        return false
    else
        aux = @head.value
        @head = @head.next
        return aux
    end
end

def extract_tail
    aux=""
    if(@tail == nil)
        return false
    else
        aux = @tail.value
        @tail = @tail.prev
        return aux
    end
end

def insert(value)
		node = Node.new(value,nil)
		if (@head == nil)
			@head = node
		else
			@tail.next = node
			node.prev = @tail
		end

		@tail=node
end
